69void OscilloscopeAudioProcessorEditor::resized()
70{
71 // This is generally where you'll want to lay out the positions of any
72 // subcomponents in your editor..
73 auto area = getLocalBounds();
74 int height = area.getHeight();
75 int width = area.getWidth();
76 float yDiff = (float)(height - 30) - oscilloscopeComponent->getHeight();
77
78 // rescale oscilloscope
79 float x_sc = (float)width / oscilloscopeComponent->getWidth();
80 float y_sc = (float)(height - 30) / oscilloscopeComponent->getHeight();
81 oscilloscopeComponent->setTransform(juce::AffineTransform::scale(x_sc, y_sc));
82 drawGrid.setTransform(juce::AffineTransform::translation(0, yDiff));
83
84 audioProcessor.storeEditorSize(getWidth(), getHeight());
85
86}
