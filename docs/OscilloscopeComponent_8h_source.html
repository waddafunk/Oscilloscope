192 void timerCallback() override
193 {
194 audioProcessor.getAudioBufferQueue()->pop(notInterpolatedData.data());
195 int queueSize = newlyPopped.size();
196 interpolator.process(ratio, notInterpolatedData.data(), newlyPopped.data(), queueSize);
197 std::copy(sampleData.data() + queueSize, sampleData.data() + sampleData.size(), newData.begin());
198 std::copy(newlyPopped.data(), newlyPopped.data() + queueSize, newData.begin() + sampleData.size() - queueSize);
199 sampleData = newData;
200 repaint();
201 }
