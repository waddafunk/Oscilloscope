38 void process(const SampleType* data, size_t numSamples)
39 {
40 size_t index = 0;
41
42 if (state == State::waitingForTrigger)
43 {
44 // If no sample over treshold (silence) flush everything.
45 if (std::all_of(data, data + numSamples, [](SampleType i) {return i < triggerLevel; }))
46 {
47 audioBufferQueue.flush();
48 return;
49 }
50 // Else setup collecting stage
51 else
52 {
53 auto result = ranges::find_if(data, data + numSamples, [](SampleType i) {return i > triggerLevel; });
54 prevSample = *result;
55 numCollected = 0;
56 state = State::collecting;
57 index = result - data;
58 }
59 }
60
61 if (state == State::collecting)
62 {
63 while (index++ < numSamples)
64 {
65 buffer[numCollected++] = *data++;
66
67 if (numCollected == buffer.size())
68 {
69 audioBufferQueue.push(buffer.data(), buffer.size());
70 state = State::waitingForTrigger;
71 prevSample = SampleType(100);
72 break;
73 }
74 }
75 }
76 }
