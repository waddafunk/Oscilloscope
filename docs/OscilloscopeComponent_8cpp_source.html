13OscilloscopeComponent::OscilloscopeComponent(OscilloscopeAudioProcessor& aProcessor, int sampleRate, int framesPerSecond) :
14 audioProcessor(aProcessor)
15{
16 setFramesPerSecond(framesPerSecond);
17 this->sampleRate = sampleRate;
18 displayLength = (int)(*aProcessor.getTreeState()->getRawParameterValue("bufferLength") * aProcessor.getSampleRate());
19 ratio = (double)displayLength / (double)EDITOR_INITIAL_WIDTH();

113void OscilloscopeComponent::timerCallback()
114{
115 audioProcessor.getAudioBufferQueue()->pop(notInterpolatedData.data());
116 int queueSize = newlyPopped.size();
117 interpolator.process(ratio, notInterpolatedData.data(), newlyPopped.data(), queueSize);
118 std::copy(sampleData.data() + queueSize, sampleData.data() + sampleData.size(), newData.begin());
119 std::copy(newlyPopped.data(), newlyPopped.data() + queueSize, newData.begin() + sampleData.size() - queueSize);
120 sampleData = newData;
121 subclassSpecificCallback();
122 repaint();
123}
